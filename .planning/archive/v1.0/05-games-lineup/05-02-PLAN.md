```typescript
import { IsArray, IsString, IsNotEmpty, IsOptional, IsIn, ValidateNested, ArrayMaxSize } from 'class-validator';
import { Type } from 'class-transformer';

export class SaveLineupEntryDto {
  @IsString()
  @IsNotEmpty()
  playerId: string;

  @IsString()
  @IsOptional()
  positionName?: string;

  @IsIn(['starting', 'bench'])
  status: 'starting' | 'bench';
}

export class SaveLineupDto {
  @IsArray()
  @ArrayMaxSize(11)
  @ValidateNested({ each: true })
  @Type(() => SaveLineupEntryDto)
  entries: SaveLineupEntryDto[];
}
```

Note: `class-transformer` is already in the project's package.json. Do NOT add any new npm dependencies.

|-----------|----------|-----------|-------------|-----------------|
| T-5-02-01 | Tampering | CreateGameDto scheduledAt | mitigate | `@IsDateString()` validator rejects non-ISO8601 strings — prevents malformed timestamps from reaching TypeORM |
| T-5-02-02 | DoS | SaveLineupDto entries | mitigate | `@ArrayMaxSize(11)` hard limit on lineup entries — prevents abnormally large arrays (e.g., 1000 entries) from reaching the bulk-replace database operation |
| T-5-02-03 | Tampering | SaveLineupEntryDto status | mitigate | `@IsIn(['starting', 'bench'])` — only valid status values accepted; prevents injection of arbitrary status strings into the lineup_entries table |
